# Create a new customer
curl -X POST YOUR_AZURE_URL/api/users \
  -H "Content-Type: application/json" \
  -d '{"username":"kate","password":"123456","role":"customer"}'

# Create a staff member
curl -X POST YOUR_AZURE_URL/api/users \
  -H "Content-Type: application/json" \
  -d '{"username":"staff1","password":"123456","role":"staff"}'

# List all users (clean format)
curl YOUR_AZURE_URL/api/users

# List only customers
curl YOUR_AZURE_URL/api/users?role=customer

# List only admin/staff
curl YOUR_AZURE_URL/api/users?role=admin

# Change password
curl -X PATCH YOUR_AZURE_URL/api/users/username/kate \
  -H "Content-Type: application/json" \
  -d '{"password":"newpass999"}'

# Change role
curl -X PATCH YOUR_AZURE_URL/api/users/username/kate \
  -H "Content-Type: application/json" \
  -d '{"role":"staff"}'

# Delete a user
curl -X DELETE YOUR_AZURE_URL/api/users/username/kate
